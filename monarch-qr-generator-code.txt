        function createQRPayload(reward) {
            const payload = {
                type: 'monarch_reward',
                rewardId: reward.reward_id,
                season: reward.season,
                timestamp: reward.expired ? Date.now() - 86400000 - 3600000 : Date.now()
            };

        }
